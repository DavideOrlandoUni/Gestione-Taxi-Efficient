 31# definizione euristica Manhattan
 32def manhattan(a: Coord, b: Coord) -> int:
 33    """Distanza di Manhattan tra due celle (movimento ortogonale)."""
 34    return abs(a[0] - b[0]) + abs(a[1] - b[1])
 35
 36# definizione algoritmo A*
 37def find_path(start: Coord, goal: Coord) -> List[Coord]:
 38
 39    if start == goal:
 40        return [start]
 41
 42    openq: List[Tuple[int, Coord]] = []
 43    heapq.heappush(openq, (0, start))
 44
 45    came_from: Dict[Coord, Optional[Coord]] = {start: None}
 46    g_score: Dict[Coord, int] = {start: 0}
 47
 48    while openq:
 49        _, current = heapq.heappop(openq)
 50
 51        if current == goal:
 52            #ricostruisci il percorso retrocedendo con came_from
 53            path: List[Coord] = [goal]
 54            while came_from[path[-1]] is not None:
 55                path.append(came_from[path[-1]])
 56            path.reverse()
 57            return path[:-1]  # esclude goal
 58
 59        x, y = current
 60        for nx, ny in ((x + 1, y), (x - 1, y), (x, y + 1), (x, y - 1)):
 61            if 0 <= nx < GRID_W and 0 <= ny < GRID_H:
 62                neigh = (nx, ny)
 63                if neigh in OBSTACLES:
 64                    continue
 65
 66                tentative_g = g_score[current] + 1
 67                if neigh not in g_score or tentative_g < g_score[neigh]:
 68                    g_score[neigh] = tentative_g
 69                    f = tentative_g + manhattan(neigh, goal)
 70                    came_from[neigh] = current
 71                    heapq.heappush(openq, (f, neigh))
 72
 73    #nessun percorso trovato
 74    return []

 79def _first_existing(cands: List[Path]) -> Optional[str]:
 80    for p in cands:
 81        if p.exists() and p.is_file():
 82            return str(p)

 85def read_plan(filepath: str) -> List[str]:
 86    """Legge un piano generato da Fast Downward e restituisce solo le azioni tra parentesi."""
 87    actions: List[str] = []
 88    with open(filepath, "r", encoding="utf-8") as f:
 89        for raw in f:
 90            s = raw.strip()
 91            if not s or s.startswith(";"):
 92                continue
 93            if "(" in s and ")" in s:
 94                s = s[:s.rfind(")") + 1]
 95            if s.startswith("(") and s.endswith(")"):
 96                actions.append(s.lower())
 97    if not actions:
 98        raise ValueError(f"Nessuna azione valida nel piano: {filepath}")
 99    return actions

101def load_locations(filepath: str) -> Dict[str, Coord]:
102    """Carica file location da JSON"""
103    with open(filepath, "r", encoding="utf-8") as f:
104        data = json.load(f)
105
106    locs: Dict[str, Coord] = {}
107    for k, v in data.items():
108        if k.lower() == "st":
109            continue
110        locs[k.lower()] = (int(v[0]), int(v[1]))
111    locs["st"] = STATION
112    return locs

126def parse_sas_plan_to_trip(actions: List[str], locs: Dict[str, Coord]) -> Tuple[TripPlan, Dict[str, Coord]]:
127    """Traduce azioni SAS (move/pickup/dropoff) in TripPlan"""
128    full: List[Coord] = []
129    picks: Dict[int, List[str]] = {}
130    drops: Dict[int, List[str]] = {}
131    labels: Dict[str, Coord] = {}
132    current: Optional[Coord] = None
133
134    def coord_of(label: str) -> Coord:
135        key = label.lower()
136        if key not in locs:
137            raise KeyError(f"Location '{label}' non definita nel file locations.")
138        return locs[key]
139
140    for raw in actions:
141        tokens = raw.strip("()").split()
142        if not tokens:
143            continue
144        op = tokens[0]
145
146        if op == "move":
147            _, _taxi, src, dst = tokens
148            src_c = coord_of(src)
149            dst_c = coord_of(dst)
150            if current is None:
151                current = src_c
152                full.append(current)
153            seg = find_path(current, dst_c)
154            full += seg
155            full.append(dst_c)
156            current = dst_c
157
158        elif op == "pickup":
159            _, _taxi, p, l = tokens
160            label = p.upper()
161            labels[label] = coord_of(l)
162            if not full:
163                current = coord_of(l)
164                full.append(current)
165            picks.setdefault(len(full) - 1, []).append(label)
166
167        elif op == "dropoff":
168            _, _taxi, p, _l = tokens
169            label = p.upper()
170            if not full:
171                current = STATION
172                full.append(current)
173            drops.setdefault(len(full) - 1, []).append(label)
174
175    trip = TripPlan(full, picks, drops)
176    return trip, labels

178def extract_pickups(actions: List[str]) -> Dict[str, str]:
179    """Estrae dal piano la prima associazione passeggero->location (es. 'P1'->'l1')."""
180    mapping: Dict[str, str] = {}
181    for raw in actions:
182        t = raw.strip("()").split()
183        if len(t) == 4 and t[0] == "pickup":
184            p = t[2].upper()
185            l = t[3].lower()
186            if p not in mapping:
187                mapping[p] = l
188    return mapping

190def build_share_two_closest_trip(pass_to_loclabel: Dict[str, str], locs: Dict[str, Coord]) -> Tuple[TripPlan, Dict[str, Coord]]:
191    """Costruisce un TripPlan che preleva i due passeggeri pi vicini tra loro ed una seconda corsa per il rimanente"""
192    # Prendi solo i passeggeri che hanno una location definita
193    items = [(p, pass_to_loclabel[p]) for p in sorted(pass_to_loclabel.keys()) if pass_to_loclabel[p] in locs]
194    if len(items) < 2:
195        # troppo pochi per lo share  costruisci un trip banale fedele al piano
196        raise ValueError("Per il ricalcolo automatico servono almeno due pickup nel piano.")
197
198    # Costruisci lista [(label, coord)]
199    pts: List[Tuple[str, Coord]] = [(p, locs[l]) for p, l in items]
200    labels_map: Dict[str, Coord] = {p: c for p, c in pts}
201
202    # Trova la coppia pi vicina (Manhattan)
203    best_pair = None
204    best_d = None
205    for i in range(len(pts)):
206        for j in range(i + 1, len(pts)):
207            d = manhattan(pts[i][1], pts[j][1])
208            if best_pair is None or d < best_d:
209                best_pair = (pts[i], pts[j])  # ((p,coord),(p,coord))
210                best_d = d
211    assert best_pair is not None
212    (pA, cA), (pB, cB) = best_pair
213
214    # Il resto (se sono 3)  il "solo"
215    remaining = [p for p, _ in pts if p not in (pA, pB)]
216
217    # Ordina i due del carpool: prima quello pi vicino alla stazione
218    first, second = (pA, cA), (pB, cB)
219    if manhattan(cB, STATION) < manhattan(cA, STATION):
220        first, second = (pB, cB), (pA, cA)
221
222    full: List[Coord] = []
223    picks: Dict[int, List[str]] = {}
224    drops: Dict[int, List[str]] = {}
225
226    # Corsa 1: ST -> first -> second -> ST
227    seg = find_path(STATION, first[1])
228    full += seg
229    full.append(first[1]); picks[len(full) - 1] = [first[0]]
230    seg = find_path(first[1], second[1])
231    full += seg
232    full.append(second[1]); picks.setdefault(len(full) - 1, []).append(second[0])
233    seg = find_path(second[1], STATION)
234    full += seg
235    full.append(STATION); drops[len(full) - 1] = [first[0], second[0]]
236
237    # Corsa 2: per ogni rimanente (di solito 1)
238    for p in remaining:
239        c = labels_map[p]
240        seg = find_path(STATION, c)
241        full += seg
242        full.append(c); picks[len(full) - 1] = [p]
243        seg = find_path(c, STATION)
244        full += seg
245        full.append(STATION); drops.setdefault(len(full) - 1, []).append(p)
246
247    trip = TripPlan(full, picks, drops)
248    return trip, labels_map

332    def _load_from_files(self, plan_candidates: List[Path], loc_candidates: List[Path], title: str, auto_share_pair: bool):
333        plan_path = _first_existing(plan_candidates)
334        if not plan_path:
335            print(f"[ERRORE] Nessun file piano trovato tra: {', '.join(map(str, plan_candidates))}")
336            return
337
338        loc_path = _first_existing(loc_candidates)
339        if not loc_path:
340            print(f"[ERRORE] File locations non trovato tra: {', '.join(map(str, loc_candidates))}")
341            return
342
343        try:
344            actions = read_plan(plan_path)
345            locs = load_locations(loc_path)
346
347            if auto_share_pair:
348                # Ricava dal piano quali passeggeri e location esistono e poi ricostruisce l'ordine corretto in base alle posizioni correnti.
349                mapping = extract_pickups(actions)
350                trip, labels = build_share_two_closest_trip(mapping, locs)
351            else:
352                trip, labels = parse_sas_plan_to_trip(actions, locs)
353
354        except Exception as e:
355            print(f"[ERRORE] Impossibile caricare piano/locations: {e}")
356            return
357
358        #aggiorna stato GUI
359        self.root.title(f"Taxi Planner - {title}")
360        self.plan, self.labels = trip, labels
361        self.reset_costs()
362        self.redraw_scene()
363        print(f"[LOG] Caricato {title} | plan={plan_path} | locations={loc_path} | auto_share={auto_share_pair}")

467    def update_cost_step(self, idx: int):
468        """Aggiorna la ripartizione dei costi."""
469        if not self.plan:
470            return
471        onboard: List[str] = []
472        for i in range(idx + 1):
473            if i in self.plan.pick_events:
474                for c in self.plan.pick_events[i]:
475                    if c not in onboard:
476                        onboard.append(c)
477            if i in self.plan.drop_events:
478                for c in self.plan.drop_events[i]:
479                    if c in onboard:
480                        onboard.remove(c)
481        if onboard and idx < len(self.plan.path) - 1:
482            self.total_cost += 1
483            share = 1.0 / len(onboard)
484            for c in onboard:
485                self.fare[c] = round(self.fare.get(c, 0.0) + share, 2)
486            print(f"[LOG] Step {idx}: passeggeri a bordo={onboard}, costo totale={self.total_cost}")
487        self.cost_total_var.set(f"Totale: {self.total_cost}")
488        self.render_cost_rows()

490    def step(self):
491        if not self.plan or not self.plan.path or self.taxi_id is None:
492            return
493        if self.path_idx >= len(self.plan.path) - 1:
494            self.is_playing = False
495            self.btn_play.config(text="Play ")
496            print("[LOG] Fine del percorso.")
497            return
498        self.path_idx += 1
499        x1, y1, x2, y2 = self.cell_to_xy(self.plan.path[self.path_idx])
500        self.canvas.coords(self.taxi_id, x1 + 8, y1 + 8, x2 - 8, y2 - 8)
501        self.draw_trace(self.path_idx)
502        self.update_cost_step(self.path_idx - 1)
503        print(f"[LOG] Step {self.path_idx}: Taxi in {self.plan.path[self.path_idx]}")

def find_path(start: Tuple[int, int], goal: Tuple[int, int]) -> List[Tuple[int, int]]: 
38def find_path(start: Coord, goal: Coord) -> List[Coord]:
39
40    if start == goal:
41        return [start]
42
43    openq: List[Tuple[int, Coord]] = []
44    heapq.heappush(openq, (0, start))
45
46    came_from: Dict[Coord, Optional[Coord]] = {start: None}
47    g_score: Dict[Coord, int] = {start: 0}
48
49    while openq:
50        _, current = heapq.heappop(openq)
51
52        if current == goal:
53            #ricostruisci il percorso retrocedendo con came_from
54            path: List[Coord] = [goal]
55            while came_from[path[-1]] is not None:
56                path.append(came_from[path[-1]])
57            path.reverse()
58            return path[:-1]  # esclude goal
59
60        x, y = current
61        for nx, ny in ((x + 1, y), (x - 1, y), (x, y + 1), (x, y - 1)):
62            if 0 <= nx < GRID_W and 0 <= ny < GRID_H:
63                neigh = (nx, ny)
64                if neigh in OBSTACLES:
65                    continue
66
67                tentative_g = g_score[current] + 1
68                if neigh not in g_score or tentative_g < g_score[neigh]:
69                    g_score[neigh] = tentative_g
70                    f = tentative_g + manhattan(neigh, goal)
71                    came_from[neigh] = current
72                    heapq.heappush(openq, (f, neigh))
73
74    #nessun percorso trovato
75    return []

def build_share_two_closest_trip( pass_to_loclabel: Dict[str, str], locs: Dict[str, Tuple[int, int]]) -> Tuple[TripPlan, Dict[str, Tuple[int, int]]]: 
191def build_share_two_closest_trip(pass_to_loclabel: Dict[str, str], locs: Dict[str, Coord]) -> Tuple[TripPlan, Dict[str, Coord]]:
192    """Costruisce un TripPlan che preleva i due passeggeri pi vicini tra loro ed una seconda corsa per il rimanente"""
193    # Prendi solo i passeggeri che hanno una location definita
194    items = [(p, pass_to_loclabel[p]) for p in sorted(pass_to_loclabel.keys()) if pass_to_loclabel[p] in locs]
195    if len(items) < 2:
196        # troppo pochi per lo share  costruisci un trip banale fedele al piano
197        raise ValueError("Per il ricalcolo automatico servono almeno due pickup nel piano.")
198
199    # Costruisci lista [(label, coord)]
200    pts: List[Tuple[str, Coord]] = [(p, locs[l]) for p, l in items]
201    labels_map: Dict[str, Coord] = {p: c for p, c in pts}
202
203    # Trova la coppia pi vicina (Manhattan)
204    best_pair = None
205    best_d = None
206    for i in range(len(pts)):
207        for j in range(i + 1, len(pts)):
208            d = manhattan(pts[i][1], pts[j][1])
209            if best_pair is None or d < best_d:
210                best_pair = (pts[i], pts[j])  # ((p,coord),(p,coord))
211                best_d = d
212    assert best_pair is not None
213    (pA, cA), (pB, cB) = best_pair
214
215    # Il resto (se sono 3)  il "solo"
216    remaining = [p for p, _ in pts if p not in (pA, pB)]
217
218    # Ordina i due del carpool: prima quello pi vicino alla stazione
219    first, second = (pA, cA), (pB, cB)
220    if manhattan(cB, STATION) < manhattan(cA, STATION):
221        first, second = (pB, cB), (pA, cA)
222
223    full: List[Coord] = []
224    picks: Dict[int, List[str]] = {}
225    drops: Dict[int, List[str]] = {}
226
227    # Corsa 1: ST -> first -> second -> ST
228    seg = find_path(STATION, first[1])
229    full += seg
230    full.append(first[1]); picks[len(full) - 1] = [first[0]]
231    seg = find_path(first[1], second[1])
232    full += seg
233    full.append(second[1]); picks.setdefault(len(full) - 1, []).append(second[0])
234    seg = find_path(second[1], STATION)
235    full += seg
236    full.append(STATION); drops[len(full) - 1] = [first[0], second[0]]
237
238    # Corsa 2: per ogni rimanente (di solito 1)
239    for p in remaining:
240        c = labels_map[p]
241        seg = find_path(STATION, c)
242        full += seg
243        full.append(c); picks[len(full) - 1] = [p]
244        seg = find_path(c, STATION)
245        full += seg
246        full.append(STATION); drops.setdefault(len(full) - 1, []).append(p)
247
248    trip = TripPlan(full, picks, drops)
249    return trip, labels_map

def update_cost_step(self, idx: int): 
468    def update_cost_step(self, idx: int):
469        """Aggiorna la ripartizione dei costi."""
470        if not self.plan:
471            return
472        onboard: List[str] = []
473        for i in range(idx + 1):
474            if i in self.plan.pick_events:
475                for c in self.plan.pick_events[i]:
476                    if c not in onboard:
477                        onboard.append(c)
478            if i in self.plan.drop_events:
479                for c in self.plan.drop_events[i]:
480                    if c in onboard:
481                        onboard.remove(c)
482        if onboard and idx < len(self.plan.path) - 1:
483            self.total_cost += 1
484            share = 1.0 / len(onboard)
485            for c in onboard:
486                self.fare[c] = round(self.fare.get(c, 0.0) + share, 2)
487            print(f"[LOG] Step {idx}: passeggeri a bordo={onboard}, costo totale={self.total_cost}")
488        self.cost_total_var.set(f"Totale: {self.total_cost}")
489        self.render_cost_rows()
